const TMDB_IMAGE_BASE = 'https://image.tmdb.org/t/p/w500';
const DEFAULT_POSTER = '/images/no-poster.png';

// ============================================
// IMAGE URL HELPER
// ============================================
function getImageUrl(imageObj: any): string {
  // Handle null/undefined
  if (!imageObj) return DEFAULT_POSTER;
  
  // If it's a string
  if (typeof imageObj === 'string') {
    if (!imageObj || imageObj === '') return DEFAULT_POSTER;
    
    // Full URL
    if (imageObj.startsWith('http://') || imageObj.startsWith('https://')) {
      return imageObj;
    }
    
    // TMDB path
    if (imageObj.startsWith('/') && !imageObj.startsWith('//')) {
      // Check if it's a local file path
      if (imageObj.startsWith('/images/')) {
        return imageObj; // Local image
      }
      return `${TMDB_IMAGE_BASE}${imageObj}`;
    }
    
    return imageObj;
  }
  
  // If it's an object with nested structure
  if (typeof imageObj === 'object' && imageObj !== null) {
    // Priority: url > fallback > path > source
    let url = null;
    
    if (imageObj.url) {
      url = imageObj.url;
    } else if (imageObj.fallback) {
      url = imageObj.fallback;
    } else if (imageObj.path) {
      url = imageObj.path;
    } else if (imageObj.source && typeof imageObj.source === 'string') {
      url = imageObj.source;
    }
    
    // If we found a URL, recursively process it
    if (url) {
      return getImageUrl(url);
    }
  }
  
  // Final fallback
  return DEFAULT_POSTER;
}

export default function HeroDetailClient({ heroData: initialHeroData }: { heroData: any }) {
  const { data: session } = useSession();
  const router = useRouter();

  const [heroData] = useState<any>(initialHeroData);
  const [isFollowing, setIsFollowing] = useState(false);
  const [followerCount, setFollowerCount] = useState(0);
  const [activeTab, setActiveTab] = useState('overview');

  useEffect(() => {
    const loadFollowData = async () => {
      if (!heroData) return;
      try {
        if (session?.user) {
          const followStatus = await getHeroFollowStatus(heroData.slug);
          setIsFollowing(followStatus);
        }
        const count = await getHeroFollowerCount(heroData.slug);
        setFollowerCount(count);
      } catch (error) {
        console.error('Error loading follow data:', error);
      }
    };
    loadFollowData();
  }, [heroData, session]);

  const handleFollow = async () => {
    if (!session?.user) {
      router.push('/signin');
      return;
    }
    try {
      await toggleHeroFollow(heroData.slug, heroData.name);
      setIsFollowing(!isFollowing);
      setFollowerCount((prev) => (isFollowing ? prev - 1 : prev + 1));
    } catch (error) {
      console.error('Error toggling follow:', error);
    }
  };

  if (!heroData) {
    return (
      <div className="min-h-screen bg-black flex items-center justify-center">
        <motion.p className="text-white text-lg">❌ Hero not found</motion.p>
      </div>
    );
  }

  return (
    <main className="min-h-screen bg-black text-white">
      {/* NAVBAR */}
      <motion.nav
        initial={{ opacity: 0, y: -20 }}
        animate={{ opacity: 1, y: 0 }}
        className="sticky top-0 z-50 bg-black/95 backdrop-blur-xl border-b border-white/10"
      >
        <div className="max-w-7xl mx-auto px-4 sm:px-6 md:px-8 py-3 md:py-4 flex items-center justify-between gap-2">
          <div className="flex items-center gap-2 md:gap-4">
            <motion.button
              onClick={() => router.back()}
              whileHover={{ scale: 1.1 }}
              whileTap={{ scale: 0.95 }}
              className="p-2 hover:bg-white/10 rounded-lg transition"
            >
              <FaArrowLeft size={18} />
            </motion.button>
            <Link href="/heroes" className="text-lg md:text-xl font-black hidden sm:block">
              🎬 Heroes
            </Link>
          </div>

          <h1 className="text-base md:text-lg font-bold truncate text-center flex-1">
            {heroData?.name || 'Unknown'}
          </h1>

          <div className="flex items-center gap-2 md:gap-3 flex-shrink-0">
            <Link href="/heroes" className="px-2 md:px-3 py-1.5 md:py-2 text-xs md:text-sm text-white/70 hover:text-white hover:bg-white/10 rounded-lg transition hidden md:block">
              Heroes
            </Link>
            <motion.button
              onClick={handleFollow}
              whileHover={{ scale: 1.05 }}
              whileTap={{ scale: 0.95 }}
              className={`px-3 md:px-4 py-1.5 md:py-2 rounded-lg font-bold text-xs md:text-sm transition ${
                isFollowing
                  ? 'bg-white/20 text-white hover:bg-white/30'
                  : 'bg-white text-black hover:shadow-lg'
              }`}
            >
              {isFollowing ? <FaHeart className="inline mr-2" size={12} /> : <FaRegHeart className="inline mr-2" size={12} />}
              <span className="hidden sm:inline">{isFollowing ? 'Following' : 'Follow'}</span>
              <span className="sm:hidden">{isFollowing ? '❤️' : '🤍'}</span>
            </motion.button>
          </div>
        </div>
      </motion.nav>

      {/* HERO BANNER */}
      <HeroBanner hero={heroData} isFollowing={isFollowing} followerCount={followerCount} onFollow={handleFollow} />

      {/* TAB NAVIGATION */}
      <TabNavigation activeTab={activeTab} setActiveTab={setActiveTab} hero={heroData} />

      {/* TAB CONTENT */}
      <div className="max-w-7xl mx-auto px-4 sm:px-6 md:px-8 py-8 md:py-12">
        <AnimatePresence mode="wait">
          <motion.div
            key={activeTab}
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: -10 }}
            transition={{ duration: 0.2 }}
          >
            {activeTab === 'overview' && <OverviewTab hero={heroData} />}
            {activeTab === 'movies' && <MoviesTab movies={heroData.movies} hero={heroData} />}
            {activeTab === 'upcoming' && <UpcomingTab upcoming={heroData.upcomingProjects || heroData.upcoming} />}
            {activeTab === 'lifestyle' && <LifestyleTab hero={heroData} />}
            {activeTab === 'philanthropy' && <PhilanthropyTab hero={heroData} />}
            {activeTab === 'brands' && <BrandsTab hero={heroData} />}
            {activeTab === 'info' && <InfoTab hero={heroData} />}
            {activeTab === 'career' && <CareerTab hero={heroData} />}
            {activeTab === 'awards' && <AwardsTab awards={heroData.awards} />}
            {activeTab === 'news' && <NewsTab news={heroData.recentNews} />}
            {activeTab === 'trivia' && <TriviaTab trivia={heroData.trivia} quotes={heroData.quotes} />}
            {activeTab === 'gallery' && <GalleryTab images={heroData.images?.gallery} />}
          </motion.div>
        </AnimatePresence>
      </div>
    </main>
  );
}

function TabNavigation({ activeTab, setActiveTab, hero }: any) {
  const scrollContainerRef = useRef<HTMLDivElement>(null);

  const tabs = [
    { id: 'info', label: '👤 Info' },
    { id: 'overview', label: '📖 Overview' },
    { id: 'movies', label: `🎬 Movies (${hero.movies?.length || 0})` },
    { id: 'upcoming', label: `📅 Upcoming (${hero.upcomingProjects?.length || hero.upcoming?.length || 0})` },
    { id: 'lifestyle', label: `✨ Lifestyle` },
    { id: 'philanthropy', label: '❤️ Philanthropy' },
    { id: 'brands', label: `⭐ Brands (${hero.brandEndorsements?.length || 0})` },
    { id: 'career', label: '📈 Career' },
    { id: 'awards', label: `🏆 Awards (${hero.awards?.length || 0})` },
    { id: 'news', label: `📰 News (${hero.recentNews?.length || 0})` },
    { id: 'trivia', label: '💡 Trivia' },
    { id: 'gallery', label: `🖼️ Gallery (${hero.images?.gallery?.length || 0})` }
  ];

  // Auto-scroll to active tab
  useEffect(() => {
    const activeTabElement = scrollContainerRef.current?.querySelector(`[data-tab="${activeTab}"]`);
    if (activeTabElement) {
      activeTabElement.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'center' });
    }
  }, [activeTab]);

  // Keyboard navigation
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      const currentIndex = tabs.findIndex(t => t.id === activeTab);
      if (e.key === 'ArrowLeft' && currentIndex > 0) setActiveTab(tabs[currentIndex - 1].id);
      else if (e.key === 'ArrowRight' && currentIndex < tabs.length - 1) setActiveTab(tabs[currentIndex + 1].id);
    };
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [activeTab, tabs, setActiveTab]);

  const scroll = (direction: 'left' | 'right') => {
    if (scrollContainerRef.current) {
      scrollContainerRef.current.scrollBy({
        left: direction === 'left' ? -200 : 200,
        behavior: 'smooth'
      });
    }
  };

  return (
    <motion.div
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      className="sticky top-[72px] z-40 bg-black/95 backdrop-blur-xl border-b border-white/10 overflow-x-auto"
    >
      <div className="max-w-7xl mx-auto px-2 sm:px-4 md:px-8 relative">
        {/* Left Arrow - Hide on mobile */}
        <button
          onClick={() => scroll('left')}
          className="absolute left-0 top-1/2 -translate-y-1/2 z-10 w-8 h-full bg-gradient-to-r from-black to-transparent flex items-center justify-center text-white hover:text-white/80 transition hidden md:flex"
          aria-label="Scroll left"
        >
          <FaChevronLeft size={16} />
        </button>

        {/* Tabs */}
        <div
          ref={scrollContainerRef}
          className="flex gap-1 md:gap-2 overflow-x-auto scrollbar-hide py-2 px-0 md:px-8"
        >
          {tabs.map((tab) => (
            <motion.button
              key={tab.id}
              data-tab={tab.id}
              onClick={() => setActiveTab(tab.id)}
              whileHover={{ scale: 1.05 }}
              whileTap={{ scale: 0.95 }}
              className={`px-2 md:px-4 py-2 md:py-3 rounded-lg font-bold text-xs md:text-sm whitespace-nowrap transition ${
                activeTab === tab.id
                  ? 'bg-white text-black'
                  : 'bg-white/10 text-white/70 hover:bg-white/20 hover:text-white'
              }`}
            >
              {tab.label}
            </motion.button>
          ))}
        </div>

        {/* Right Arrow - Hide on mobile */}
        <button
          onClick={() => scroll('right')}
          className="absolute right-0 top-1/2 -translate-y-1/2 z-10 w-8 h-full bg-gradient-to-l from-black to-transparent flex items-center justify-center text-white hover:text-white/80 transition hidden md:flex"
          aria-label="Scroll right"
        >
          <FaChevronRight size={16} />
        </button>
      </div>

      
    </motion.div>
  );
}

function MoviesTab({ movies, hero }: any) {
  const [filter, setFilter] = useState('all');
  const [sortBy, setSortBy] = useState('year');

  const { allMovies, leadCount, cameoCount, voiceoverCount, childCount } = useMemo(() => {
    const all = movies || [];
    const lead = all.filter((m: any) => m.movieType?.toLowerCase() === 'lead' || m.role?.toLowerCase() === 'lead' || !m.movieType);
    const cameo = hero?.cameoAppearances || [];
    const voiceover = hero?.voiceOverAppearances || [];
    const child = hero?.childArtistMovies || [];

    return {
      allMovies: all,
      leadCount: lead.length,
      cameoCount: cameo.length,
      voiceoverCount: voiceover.length,
      childCount: child.length
    };
  }, [movies, hero]);

  const filteredMovies = useMemo(() => {
    let filtered = [];

    if (filter === 'all') {
      filtered = [...allMovies, ...(hero?.cameoAppearances || []), ...(hero?.voiceOverAppearances || []), ...(hero?.childArtistMovies || [])];
    } else if (filter === 'lead') {
      filtered = allMovies.filter((m: any) => m.movieType?.toLowerCase() === 'lead' || m.role?.toLowerCase() === 'lead' || !m.movieType);
    } else if (filter === 'cameo') {
      filtered = hero?.cameoAppearances || [];
    } else if (filter === 'voiceover') {
      filtered = hero?.voiceOverAppearances || [];
    } else if (filter === 'child') {
      filtered = hero?.childArtistMovies || [];
    }

    return filtered.sort((a: any, b: any) => {
      if (sortBy === 'year') return (b.year || 0) - (a.year || 0);
      if (sortBy === 'rating') return (b.ratings?.imdb || 0) - (a.ratings?.imdb || 0);
      return 0;
    });
  }, [allMovies, hero, filter, sortBy]);

  return (
    <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} className="space-y-6 md:space-y-8">
      {/* FILTERS - RESPONSIVE */}
      <div className="flex flex-col sm:flex-row items-start sm:items-center justify-between gap-3 md:gap-4 flex-wrap">
        <div className="flex flex-wrap gap-2">
          <FilterButton active={filter === 'all'} onClick={() => setFilter('all')}>
            All ({allMovies.length + cameoCount + voiceoverCount + childCount})
          </FilterButton>
          <FilterButton active={filter === 'lead'} onClick={() => setFilter('lead')}>
            Lead ({leadCount})
          </FilterButton>
          {childCount > 0 && (
            <FilterButton active={filter === 'child'} onClick={() => setFilter('child')}>
              Child ({childCount})
            </FilterButton>
          )}
          {cameoCount > 0 && (
            <FilterButton active={filter === 'cameo'} onClick={() => setFilter('cameo')}>
              Cameo ({cameoCount})
            </FilterButton>
          )}
          {voiceoverCount > 0 && (
            <FilterButton active={filter === 'voiceover'} onClick={() => setFilter('voiceover')}>
              Voice ({voiceoverCount})
            </FilterButton>
          )}
        </div>

        {/* Sort */}
        <select
          value={sortBy}
          onChange={(e) => setSortBy(e.target.value)}
          className="px-3 md:px-4 py-2 bg-white/10 border border-white/20 text-white rounded-lg text-xs md:text-sm outline-none focus:ring-2 focus:ring-white/30 transition"
          style={{ colorScheme: 'dark' }}
        >
          <option value="year">📅 Year</option>
          <option value="rating">⭐ Rating</option>
        </select>
      </div>

      {/* MOVIES GRID - FULLY RESPONSIVE */}
      {filteredMovies.length === 0 ? (
        <motion.div
          initial={{ opacity: 0 }}
          animate={{ opacity: 1 }}
          className="text-center py-12 md:py-20"
        >
          <p className="text-white/40 text-sm md:text-base">❌ No movies found</p>
        </motion.div>
      ) : (
        <div className="grid grid-cols-2 sm:grid-cols-3 md:grid-cols-4 lg:grid-cols-5 gap-2 md:gap-4">
          {filteredMovies.map((movie: any, idx: number) => (
            <motion.div
              key={movie.slug || movie.id || idx}
              initial={{ opacity: 0, y: 10 }}
              whileInView={{ opacity: 1, y: 0 }}
              transition={{ delay: idx * 0.02 }}
            >
              <MovieCard movie={movie} />
            </motion.div>
          ))}
        </div>
      )}
    </motion.div>
  );
}

      {/* KNOWN FOR */}
      {hero.knownFor && hero.knownFor.length > 0 && (
        <motion.section
          initial={{ opacity: 0, y: 20 }}
          whileInView={{ opacity: 1, y: 0 }}
        >
          <h2 className="text-2xl md:text-3xl font-black text-white mb-4 md:mb-6">🏆 Known For</h2>
          <div className="flex flex-wrap gap-2 md:gap-3">
            {hero.knownFor.map((item: string, i: number) => (
              <motion.span
                key={i}
                initial={{ opacity: 0, scale: 0.9 }}
                whileInView={{ opacity: 1, scale: 1 }}
                transition={{ delay: i * 0.05 }}
                className="px-3 md:px-4 py-1.5 md:py-2 bg-white/10 hover:bg-white/20 text-white border border-white/20 rounded-full font-bold text-xs md:text-sm transition"
              >
                {item}
              </motion.span>
            ))}
          </div>
        </motion.section>
      )}

      {/* BOX OFFICE STATS */}
      {hero.boxOfficeStats && (
        <motion.section
          initial={{ opacity: 0, y: 20 }}
          whileInView={{ opacity: 1, y: 0 }}
        >
          <h2 className="text-2xl md:text-3xl font-black text-white mb-4 md:mb-6">📊 Box Office</h2>
          <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
            {hero.boxOfficeStats.totalGross && (
              <div className="rounded-lg p-4 md:p-6 bg-gradient-to-br from-white/15 to-white/5 border border-white/20">
                <p className="text-white/50 text-xs md:text-sm mb-2">💰 Total Gross</p>
                <p className="text-white font-black text-xl md:text-2xl">{hero.boxOfficeStats.totalGross}</p>
              </div>
            )}
            {hero.boxOfficeStats.averagePerFilm && (
              <div className="rounded-lg p-4 md:p-6 bg-white/5 border border-white/10">
                <p className="text-white/50 text-xs md:text-sm mb-2">📈 Average</p>
                <p className="text-white font-bold text-lg md:text-xl">{hero.boxOfficeStats.averagePerFilm}</p>
              </div>
            )}
            {hero.boxOfficeStats.highestGrosser && (
              <div className="rounded-lg p-4 md:p-6 bg-white/5 border border-white/10">
                <p className="text-white/50 text-xs md:text-sm mb-2">🎯 Highest</p>
                <p className="text-white font-bold text-sm md:text-base">{hero.boxOfficeStats.highestGrosser.title}</p>
                <p className="text-white/70 text-xs md:text-sm mt-1">{hero.boxOfficeStats.highestGrosser.gross}</p>
              </div>
            )}
          </div>
        </motion.section>
      )}
    </motion.div>
  );
}
